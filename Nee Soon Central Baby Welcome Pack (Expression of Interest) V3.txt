    try{
      const res = await window.storage.get(STORAGE_KEY, true);
      return res ? JSON.parse(res.value) : [];
    }catch(e){
      return [];
    }
  }
  async function saveSubmissions(list){
    try{
      await window.storage.set(STORAGE_KEY, JSON.stringify(list), true);
      return true;
    }catch(e){
      console.error('Storage error', e);
      return false;
    }
  }

  // ---------- Eligibility logic ----------
  function monthsBetween(a,b){
    return (b.getFullYear()-a.getFullYear())*12 + (b.getMonth()-a.getMonth()) + (b.getDate()-a.getDate())/30;
  }
  function computeEligibility(edd, babyDob){
    const today = new Date();
    if(babyDob){
      const dob = new Date(babyDob);
      const ageMonths = monthsBetween(dob, today);
      return { eligible: ageMonths >= 0 && ageMonths <= 6, basis: 'baby age', value: Math.max(0,Math.round(ageMonths)) };
    }
    if(edd){
      const due = new Date(edd);
      const monthsToDue = monthsBetween(today, due);
      return { eligible: monthsToDue >= 0 && monthsToDue <= 6, basis: 'EDD', value: Math.max(0,Math.round(monthsToDue)) };
    }
    return { eligible: null, basis: null, value: null };
  }

  // ---------- Live timeline widget ----------
  const eddInput = document.getElementById('edd');
  const dobInput = document.getElementById('babyDob');
  const tlDot = document.getElementById('tlDot');

  function updateTimeline(){
    const edd = eddInput.value, dob = dobInput.value;
    const result = computeEligibility(edd, dob);
    if(result.eligible === null){
      tlDot.style.left = '50%';
      tlStatus.className = 'tl-status neutral';
      tlStatus.textContent = 'Enter a due date or date of birth below to check your eligibility window.';
      return;
    }
    let pct;
    if(result.basis === 'baby age'){
      pct = 50 + Math.min(result.value,6)/6*50;
      tlStatus.textContent = result.eligible

        : `Baby is about ${result.value} month(s) old — outside the 0–6 month window.`;

        : `Your due date is more than 6 months away, or has passed — outside the window.`;
    }
    tlDot.style.left = pct + '%';
    tlStatus.className = 'tl-status ' + (result.eligible ? 'eligible' : 'ineligible');
  }

    const consentEOI = document.getElementById('consentEOI').checked;
    const consentData = document.getElementById('consentData').checked;

    const requiredOk = fullName && nric && mobile && email && address && postal && consentEOI && consentData;
    if(!requiredOk){
      document.getElementById('formError').style.display='block';
      return;
    }
    document.getElementById('formError').style.display='none';

    const elig = computeEligibility(edd, babyDob);
    const submissions = await loadSubmissions();

    // duplicate detection: NRIC suffix, phone, or address

  function applyFilters(list){
    const status = document.getElementById('fStatus').value;
    const eligible = document.getElementById('fEligible').value;
    const firstTime = document.getElementById('fFirstTime').value;
    const block = document.getElementById('fBlock').value.trim().toLowerCase();

      });
    });
  }

  function escapeHtml(str){
    return (str||'').toString().replace(/[&<>"']/g, m=>({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[m]));
  }

  ['fStatus','fEligible','fFirstTime','fBlock','fSearch'].forEach(id=>{
    document.getElementById(id).addEventListener('input', renderTable);
  });

  function renderBarChart(containerId, dataObj, maxOverride){
    const container = document.getElementById(containerId);
    const entries = Object.entries(dataObj);
    if(entries.length === 0){ container.innerHTML = '<p style="font-size:12.5px;color:var(--ink-soft);">No data yet.</p>'; return; }
    const max = maxOverride || Math.max(...entries.map(e=>e[1]), 1);
    container.innerHTML = entries.map(([label,val])=>`
      <div class="bar-row">
        <div class="lbl">${escapeHtml(label)}</div>
        <div class="bar-track"><div class="bar-fill" style="width:${(val/max*100)}%"></div></div>
        <div class="val">${val}</div>
      </div>
    `).join('');
  }

  function isoWeek(dateStr){
    const d = new Date(dateStr);
    const onejan = new Date(d.getFullYear(),0,1);
    const week = Math.ceil((((d - onejan) / 86400000) + onejan.getDay()+1)/7);
    return `${d.getFullYear()}-W${week}`;
  }
  function monthLabel(dateStr){
    const d = new Date(dateStr);
    return d.toLocaleString('en-SG',{month:'short', year:'2-digit'});
  }

  function renderAnalytics(){
    const byWeek = {};
    allSubmissions.forEach(s=>{
      const wk = isoWeek(s.timestamp);
      byWeek[wk] = (byWeek[wk]||0)+1;
    });
    renderBarChart('chartByWeek', byWeek);

    const byBlock = {};
    allSubmissions.forEach(s=>{
      const b = s.block || 'Unspecified';
      byBlock[b] = (byBlock[b]||0)+1;
    });
    renderBarChart('chartByBlock', byBlock);

    const byMonth = {};
    allSubmissions.forEach(s=>{
      if(s.edd){
        const m = monthLabel(s.edd);
        byMonth[m] = (byMonth[m]||0)+1;
      }
    });
    renderBarChart('chartByMonth', byMonth);

    const firstTimeCounts = { 'First-time':0, 'Experienced':0 };
    allSubmissions.forEach(s=>{
      if(s.firstTime === 'Yes') firstTimeCounts['First-time']++;
      else firstTimeCounts['Experienced']++;
    });
    renderBarChart('chartFirstTime', firstTimeCounts);
  }
